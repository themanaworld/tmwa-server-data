function|script|SpawnMunro

        goto L_EnableMunro;
    goto L_DisableMunro;

L_EnableMunro:
    enablenpc "Munro";
    goto L_Return;

L_DisableMunro:
    disablenpc "Munro";
    goto L_Return;

L_Return:
    return;
}
