009-3,162,80,0|script|#PVPCaveBarrier|45,0,0
{
    mes "Do you want to enter the PVP cave? It costs 150GP.";
    menu
        "No.", L_Exit,
        "Yes.", L_Next;

L_Next:
    if (Zeny < 150)
        goto L_NotEnoughMoney;
    if (Zeny >= 150)
        goto L_Pay;
    goto L_Exit;

L_Exit:
    close;

L_Pay:
    set Zeny, Zeny - 150;
    warp "009-6", 36, 47;
    close;

L_NotEnoughMoney:
    mes "You don't have enough money!";
    close;

}
