function|script|VaultLogin
{
    if (##VAULT < 1) goto L_Return;

    callsub S_Exp;
    callsub S_Gold;

    return; // go back to global handler

S_Return: // this is to end execution of a sub
    return;

L_Return:
    return;

S_Exp:
    if (##ADD_LVL < 1) goto S_Return;
    getexp ##ADD_LVL, 0;
    set ##ADD_LVL, 0;
    return;

S_Gold:
    if (##ADD_GP < 1) goto S_Return;
    set Zeny, Zeny + ##ADD_GP;
    set ##ADD_GP, 0;
    return;
}
