// non-spammy news system
// authors: wushin, meko

function|script|newsHandler|,
{
  if((@nw0$ == "changelog") || (@nw0$ == "cl")) goto L_Changelog;
  if((@nw0$ == "contributors") || (@nw0$ == "cts")) goto L_Contributors;
  goto L_Usage;
  
  L_Changelog:
    callfunc "newsChangelog";
  end;
  
  L_Contributors:
    callfunc "newsContributors";
  end;
  
  L_Usage:
    mes "[News]";
    mes "";
    mes "%news changelog";
    mes "    Display what's new in this version.";
    mes "";
    mes "%news contributors";
    mes "    Display contributors of this version.";
    menu
      "Changelog", L_Changelog,
      "Contributors", L_Contributors,
      "Close", L_Close;
  
  L_Close:
    close;
}
