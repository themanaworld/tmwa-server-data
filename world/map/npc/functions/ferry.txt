// The ferry system

function|script|FerryHelp
{
    mes "\"Walk on the dock. You'll be given a chance to board the boat.\"";
    mes "\"There are 2 Ferrys, both ferrys are free to ride.\"";
    mes "\"The Main ferry travels from Argeas, Kaizei and Tonori. It docks in the major ports Hurnscald North, Nivalis and the Tulimshar Dock.\"";
    mes "\"The Candor Ferry only travels betwen Candor and the Hurnscald South Dock.\"";
    return;
}
function|script|FerryManifest
{
    mes "Which Ferry do you want to check the schedule for?";
    menu
        "World Ferry.", L_WorldFerry,
        "Candor Ferry.", L_CandorFerry;

L_CandorFerry:
    mes "This ferry goes between Candor and Hurnscald.";
    goto L_Close;

L_WorldFerry:
    mes "This ferry can take you to Nivalis (Kaizei), Tulimshar (Tonori), or Hurnscald (Argeas)";
    goto L_Close;

L_Close:
    close2;
    return;
}

function|script|BoardFerry
{
    mes "Board the ferry?";
    menu
        "Tulimshar (Tonori)", L_Tulimshar,
        "Hurnscald (Argeas)", L_Hurnscald,
        "Nivalis (Kaizei)", L_Nivalis,
        "No.", L_Return;

L_Tulimshar:
    close2;
    warp "001-1", 63, 73;
    goto L_Return;

L_Hurnscald:
    close2;
    warp "008-1", 137, 64;
    goto L_Return;

L_Nivalis:
    close2;
    warp "031-1", 95, 109;
    goto L_Return;

L_Return:
    return;
}

function|script|BoardCandorFerry
{
    mes "Board the ferry?";
    menu
        "Yes.", L_Board,
        "No.", L_Return;

L_Board:
    if (getmap() == "008-1") goto L_Candor;
    goto L_Hurnscald;

L_Hurnscald:
    close2;
    warp "008-1", 137, 75;
    goto L_Return;

L_Candor:
    close2;
    warp "029-1", 46, 109;
    goto L_Return;

L_Return:
    return;
}
