// Debug Crypts Battle

function|script|UndeadDebug
{
    goto L_Main;

L_Main:
    if (@undeaddebug == 3)
        goto L_UndeadDebugThree;
    if (@undeaddebug == 4)
        goto L_UndeadDebugFour;
    if (@undeaddebug == 5)
        goto L_UndeadDebugFive;
    goto L_Close;

L_UndeadDebugThree:
    mes "Reset your self to the various states.";
    mes "Options Limited to Time and Place.";
    menu
        "Get Ritual Items.", L_KrukanItems,
        "Nevermind.", L_Close;

L_UndeadDebugFour:
    mes "Reset your self to the various states.";
    mes "Options Limited to Time and Place.";
    menu
        "Get Ritual Items.", L_RazhaItems,
        "Nevermind.", L_Close;

L_UndeadDebugFive:
    mes "Reset your self to the various states.";
    mes "Options Limited to Time and Place.";
    menu
        "Get Ritual Items.", L_TeroganItems,
        "Nevermind.", L_Close;

L_KrukanItems:
    setarray @getitem$, "Soul", "Skull", "DarkCrystal", "Bone";
    setarray @getitem, 1, 5, 5, 5;
    callfunc "CheckInventory";
    if(@check_fail) goto L_Close;
    goto L_Main;

L_RazhaItems:
    setarray @getitem$, "Soul", "RottenRags", "UndeadEye", "UndeadEar";
    setarray @getitem, 3, 5, 5, 5;
    callfunc "CheckInventory";
    if(@check_fail) goto L_Close;
    goto L_Main;

L_TeroganItems:
    setarray @getitem$, "Soul", "DiseasedHeart", "UndeadEye", "UndeadEar";
    setarray @getitem, 5, 5, 5, 5;
    callfunc "CheckInventory";
    if(@check_fail) goto L_Close;
    goto L_Main;

L_Close:
    close;
}

// Debug for Krukan
027-3.gat,84,89,0|script|UndeadDebug3|155
{
    set @undeaddebug, 3;
    callfunc "UndeadDebug";
    end;
OnInit:
    if (!debug)
        disablenpc "UndeadDebug3";
    end;
}
