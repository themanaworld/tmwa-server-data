function|script|statusdebug|,
{

    goto L_Main;

L_Main: 
    mes "What status do you want to test?";
    menu
        "Poison.", L_Poison,
        "Flying Backpack.", L_Backpack,
        "Slow Poison.", L_SlowPoison,
        "Increase Speed.", L_Speed,
        "Haste.", L_Haste,
        "Magical Barrier.", L_MBarrier,
        "Shield.", L_Shield,
        "Power.", L_Strength,
        "Blind.", L_NI,
        "Sleep.", L_NI,
        "Freeze.", L_NI,
        "Silence.", L_NI,
        "Stun.", L_NI,
        "Curse.", L_NI,
        "Confusion.", L_NI,
        "Stone.", L_NI,
        "Bleeding.", L_NI,
        "Close.", L_Close;
    
L_NI:
    mes "This status is not working or not yet implemented";
    next;
    goto L_Main;        
        
L_Poison:
    sc_start sc_poison, 1, 20;
    mes "You are poisoned for 1 minute";
    next;
    goto L_Close;

L_Shield:
    sc_start sc_phys_shield, 1, 20;
    mes "You are shielded from physical attacks for 1 minute";
    next;
    goto L_Close;
    
    
L_SlowPoison:
    sc_start sc_slowpoison, 1, 20;
    mes "Poison slowed.";
    next;
    goto L_Close;
    
L_MBarrier:
    sc_start sc_mbarrier, 1, 20;
    mes "Magical barrier for 1 minute.";
    next;
    goto L_Close;
    
L_Haste:
    sc_start sc_haste, 1, 20;
    mes "Hasted for 1 minute.";
    next;
    goto L_Close;

L_Backpack:
    sc_start sc_flying_backpack, 1, 20;
    mes "Your backpack is flying for 1 minute";
    next;
    goto L_Close;    
    
L_Speed:
    sc_start sc_raiseattackspeed0, 10, 20;
    mes "Attack speed increased for 10 minutes.";
    next;
    goto L_Close;
    
L_Strength:
    sc_start sc_raiseattackstrength, 10, 20;
    mes "Attack strength increased for 10 minutes.";
    next;
    goto L_Close;
    
L_Close:
    close;
}
