// Confused Tree
// Author: Wushin based off melkior's bot
// Needs to listen via magic
function|script|ConfusedTree|,
{
    mes "[Confused Tree]";

    if ($@BBSMSG_STATUS)
        goto L_PlayBackMessage;
    goto L_NoPlayBack;

L_PlayBackMessage:
    mes "\"Hello, Can I help you?\"";
    menu
        "Any Recent News.", L_PlayMessage,
        "TMW Info.", L_MiniSite,
        "No, thanks.", L_Return;

L_NoPlayBack:
    mes "\"Hello, Can I help you?\"";
    menu
        "TMW Info.", L_MiniSite,
        "No, thanks.", L_Return;

L_MiniSite:
    mes "[Confused Tree]";
    mes "\"What would you like to know?\"";
    menu
        "I'm stuck in a quest.", L_Stuck,
        "I found a bug.", L_Bug,
        "How can I help develop?", L_Help,
        "What is The Mana World?", L_WhatIs,
        "Nevermind.", L_Return;

L_Stuck:
    mes "\"If you are stuck in a quest, first and foremost ask around.\"";
    mes "\"Check the Online list Button or by pressing ###keyWindowOnline; and then ask other players.\"";
    next;
    mes "\"You can also check the [@@https://wiki.themanaworld.org/index.php/Quests|Quests@@] on the wiki.\"";
    next;
    goto L_MiniSite;

L_Bug:
    mes "\"The easiest way to report a bug is by posting to the [@@https://forums.themanaworld.org/viewforum.php?f=2|Forums@@]\"";
    mes "\"Just make sure the bug doesn't exist in the client. If it is manaplus that is causing the problem. Try posting to the bug tracker at [@@http://bugs.evolonline.org/login_page.php|Manaplus Bug Tracker@@].\"";
    next;
    goto L_MiniSite;

L_Help:
    mes "\"If you would like to join the project, we would ask you read over [@@https://wiki.themanaworld.org/index.php/Joining_the_project|Joining the Project@@] and join us in [@@https://webchat.freenode.net/?channels=#themanaworld-dev|IRC@@].\"";
    next;
    goto L_MiniSite;

L_WhatIs:
    mes "\"The Mana World (TMW) is a serious effort to create an innovative free and open source MMORPG. TMW uses 2D graphics and aims to create a large and diverse interactive world. It is licensed under the GPL, making sure this game can't ever run away from you.\"";
    next;
    mes "\"Want to keep up-to-date with The Mana World happenings? Read the [@@https://forums.themanaworld.org/index.php|Forums@@] and follow us on [@@https://www.facebook.com/themanaworld|Facebook@@] and [@@https://plus.google.com/+TheManaWorldOrg/posts|Google+@@].\"";
    next;
    goto L_MiniSite;

L_PlayMessage:
    mes $@BBSMSG_MSG$;
    next;
    goto L_PlayBackMessage;

L_Return:
    return;
}
