function|script|scenariodebug|,
{

    goto L_Begin;
    
L_Begin:
    mes "What scenario would you like to test?";
    menu
        "Noob level 1.", L_Noob,
        "Close", L_Close;

L_Noob:
    mes "You are now a level 1 noob.";
    set BaseLevel, 1;
    statusup2 bStr, 9 - readparam(bStr);
    statusup2 bAgi, 2 - readparam(bAgi);
    statusup2 bVit, 9 - readparam(bVit);
    statusup2 bInt, 1 - readparam(bInt);
    statusup2 bDex, 9 - readparam(bDex);
    statusup2 bLuk, 1 - readparam(bLuk);
    resetstatus;
    next;
    goto L_Close;

L_Close:
  close;
}
