function|script|scenariodebug|,
{
// TODO: magic exp, zeny, items

    goto L_Begin;
    
L_Begin:
    mes "What scenario would you like to test?";
    mes "Make sure to @itemreset before choosing a scenario.";
    menu
        "Noob level 1.", L_Noob,
        "Noob level 15.", L_Noob15,
        "Close", L_Close;

S_ResetBase:
    resetstatus;
    statusup2 bStr, 9 - readparam(bStr);
    statusup2 bAgi, 2 - readparam(bAgi);
    statusup2 bVit, 9 - readparam(bVit);
    statusup2 bDex, 8 - readparam(bDex);
    cleararray @skilllist_name$[0], "", 8;
    cleararray @skilllist_id[0], 0, 8;
    cleararray @skilllist_count[0], 0, 8;
    setskill SKILL_EMOTE, 1;
    setskill SKILL_TRADE, 1;
    setskill SKILL_PARTY, 0;
    setskill SKILL_POOL, 0;
    setskill SKILL_MALLARDS_EYE, 0;
    setskill SKILL_BRAWLING, 0;
    setskill SKILL_SPEED, 0;
    setskill SKILL_RESIST_POISON, 0;
    setskill SKILL_ASTRAL_SOUL, 0;
    setskill SKILL_RAGING, 0;
    setskill SKILL_MAGIC, 0;
    setskill SKILL_MAGIC_LIFE, 0;
    setskill SKILL_MAGIC_WAR, 0;
    setskill SKILL_MAGIC_TRANSMUTE, 0;
    setskill SKILL_MAGIC_NATURE, 0;
    setskill SKILL_MAGIC_ASTRAL, 0;
    setskill SKILL_MAGIC_DARK, 0;
    return;
    
L_Noob:
    mes "You are now a level 1 noob.";
    set BaseLevel, 1;
    set Zeny, 30;
    callsub S_ResetBase;
    goto L_Close;

L_Noob15:
    mes "You are now a level 15 noob.";
    set BaseLevel, 15;
    callsub S_ResetBase;
    goto L_Close;    
    
L_Close:
  close;
}
