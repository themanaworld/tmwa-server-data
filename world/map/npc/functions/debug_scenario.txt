function|script|scenariodebug|,
{
// TODO: magic exp, zeny, items

    goto L_Begin;
    
L_Begin:
    mes "[Scenario Debug]";
    mes "Please choose the type of scenario.";
    menu
        "Noob (level < 45).", L_Noob,
        "Warrior.", L_Warrior,
        "Archer.", L_Archer,
        "Mage.", L_Mage,
        "Specialized (level 85+).", L_Spec,
        "Community builds.", L_CO,
        "Close.", L_Close;
        
L_Noob:
    menu
        "Noob level 1.", L_Noob1,
        "Noob level 15.", L_Noob15,
        "Back.", L_Begin;
        
L_Warrior:
    menu
        "Back.", L_Begin;

L_Archer:
    menu
        "Archer level 60.", L_Archer60,
        "Back.", L_Begin;
        
L_Mage:
    menu
        "Back.", L_Begin;
        
L_Spec:
    menu
        "Back.", L_Begin;

L_CO:
    mes "Community builds are builds submitted by players on the forums.";
    mes "At the moment this category is only a placeholder.";
    menu
      "Back.", L_Begin;
      
L_Noob1:
    mes "You are now a level 1 noob.";
    mes "Make sure to equip your items.";
    callsub S_ResetBase;
    getitem "CottonShirt", 1;
    getitem "Knife", 1;
    getitem "SlingShot", 1;
    getitem "SlingBullet", 100;
    getitem "RaggedShorts", 1;
    goto L_Close;

L_Noob15:
    mes "You are now a level 15 noob.";
    mes "Make sure to equip your items.";
    callsub S_ResetBase;
    set BaseLevel, 15;
    set Zeny, 1500;
    getitem "CottonShirt", 1;
    getitem "Knife", 1;
    getitem "Bow", 1;
    getitem "Arrow", 100;
    goto L_Close;

L_Archer60:
    mes "You are now a level 60 archer.";
    mes "Make sure to equip your items.";
    callsub S_ResetBase;
    callsub S_AddMagic;
    callsub S_AddMagic_Life;
    callsub S_AddPool;
    callsub S_AddPool_Mallard;
    set BaseLevel, 60;
    set Zeny, 40500;
    poolskill 45; //focus on mallard
    statusup2 bStr, 1 - readparam(bStr);
    statusup2 bAgi, 1 - readparam(bAgi);
    statusup2 bVit, 3 - readparam(bVit);
    statusup2 bDex, 92 - readparam(bDex);
    getitem "ForestBow", 1;
    getitem "FairyHat", 1;
    getitem "ForestArmor", 1;
    getitem "JeansChaps", 1;
    getitem "SilkGloves", 1;
    getitem "FurBoots", 1;
    getitem "FourLeafClover", 1;
    getitem "CatEars", 1;
    getitem "Lifestone", 100;
    getitem "Arrow", 3000;
    goto L_Close;


    
S_AddPool: //FIXME: add luca flags
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_KNOWS_UNFOCUS_RECIPE;
    setskill SKILL_POOL, 1;
    setskill SKILL_BRAWLING, 1;
    setskill SKILL_PARTY, 2;
    return;
    
S_AddPool_Mallard: //FIXME: add gwendolynn flags
    setskill SKILL_MALLARDS_EYE, 9;
    return;
    
S_AddMagic:
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_KNOWS_MANASEED;
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_DRANK_POTION;
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_TOUCHED_MANASEED;
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_KNOWS_MANAPOTION;
    set MAGIC_FLAGS, MAGIC_FLAGS | MFLAG_MANASEED_RUMOUR;
    setskill SKILL_MAGIC, 2;
    return;
    
S_AddMagic_Life: //FIXME: add elanore flags to QUEST_MAGIC2
    setskill SKILL_MAGIC_LIFE, 2;
    return;
    
S_ResetBase:
    gmcommand "@charwipe " + strcharinfo(0);
    statusup2 bStr, 9 - readparam(bStr);
    statusup2 bAgi, 2 - readparam(bAgi);
    statusup2 bVit, 9 - readparam(bVit);
    statusup2 bDex, 8 - readparam(bDex);
    setskill SKILL_EMOTE, 1;
    setskill SKILL_TRADE, 1;
    delitem "CottonShirt", 1; //charwipe leftover
    delitem "Knife", 1; //charwipe leftover
    getitem "HitchhikersTowel", 1;
    return;    
    
L_Close:
    close;
}
