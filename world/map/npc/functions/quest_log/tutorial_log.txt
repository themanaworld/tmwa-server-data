function|script|TutorialQuestLog|,
{
    // Starting Area Tutorial
    callfunc "TutorialState";
    if ((@tutorial >= 8) || (FLAGS & FLAG_TUTORIAL_DONE)) goto L_Done;
    if (@tutorial == 7) goto L_Stats;
    if (@tutorial == 6) goto L_Fail;
    if (@tutorial == 5) goto L_Maggots;
    if (@tutorial == 4) goto L_Tanisha;
    if (@tutorial == 3) goto L_Clothes;
    if (@tutorial == 2) goto L_Chest;
    if (@tutorial == 1) goto L_Carpet;
    goto L_NotStarted;

L_Done:
    mes "Complete.";
    goto L_Return;

L_Stats:
    mes "Talk to Tanisha.";
    goto L_Return;

L_Fail:
    mes "Talk to Tanisha.";
    goto L_Return;

L_Maggots:
    mes "Kill the Maggots for Tanisha.";
    goto L_Return;

L_Tanisha:
    mes "Talk to Tanisha.";
    goto L_Return;

L_Clothes:
    mes "Put on Clothes.";
    goto L_Return;

L_Chest:
    mes "Open Chest.";
    goto L_Return;

L_Carpet:
    mes "Walk on Carpet.";
    goto L_Return;

L_NotStarted:
    mes "Talk to Sorfina.";
    goto L_Return;

L_Return:
    return;
}
