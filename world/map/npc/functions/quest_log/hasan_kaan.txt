function|script|Hasan and KaanQuestLog|,
{
    callfunc "TutorialState";

    if ((getcharid(3) == $@ScorpionFighter) || ($@ScorpionFighter && $@ScorpionFighter != getcharid(3)))
        goto L_Afraid;
    if ((@hasan_complete) && ((@tutorial == 12) || (@tutorial == 14)))
        goto L_TutDone;
    if ((@hasan_complete) && (@tutorial == 13))
        goto L_Hasan_Saved;
    if (@tutorial == 12)
        goto L_Trick2;
    if ((@tutorial == 11) && ((@sorfina_hasan) || (@tanisha_hasan)))
        goto L_TrickKnown;
    if (@tutorial == 11)
        goto L_Trick1;
    if (@tutorial == 10)
        goto L_Hasan;
    goto L_Return;

L_Afraid:
    mes "Kill the scorpion.";
    goto L_Return;

L_TutDone:
    mes "Complete.";
    goto L_Return;

L_Hasan_Saved:
    mes "Complete. You saved Hasan!";
    goto L_Return;

L_Trick2:
    mes "Trick Hasan with Kaan.";
    goto L_Return;

L_TrickKnown:
    mes "Trick Hasan with Kaan.";
    goto L_Return;

L_Trick1:
    mes "Kaan needs to know Hasans weakness, talk to Tanisha or Sorfina.";
    goto L_Return;

L_Hasan:
    mes "Do something about Hasan.";
    goto L_Return;

L_Return:
    return;
}
