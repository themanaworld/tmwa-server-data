function|script|Barrels o funQuestLog|,
{
    callfunc "CheckBarrelQuest";

    if (@barrel_quest <= 2)
        goto L_Find;
    if (@barrel_quest >= 3)
        goto L_QuestEnd;
    goto L_Meet;

L_Find:
    mes "Find the Bug bomb in the store house barrels.";
    goto L_Return;

L_QuestEnd:
    mes "Complete.";
    goto L_Return;

L_Meet:
    mes "Zegas needs help.";
    goto L_Return;

L_Return:
    return;
}
