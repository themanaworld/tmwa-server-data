//The paperboy system

function|script|Paperboy|,
{
    mes "[" + @npcname$ + "]";
    mes "\"Would you like to buy one of our bi-weekly newsletters?\"";
    mes "\"Only " + @newspaper_price + " gold pieces?\"";
    menu
        "Yes I would.", L_Start_Paper,
        "No thanks.", L_Return;

L_Start_Paper:
    if (Zeny < @newspaper_price) goto L_NotEnoughMoney;
    set Zeny, Zeny - @newspaper_price;
    mes "[THE NEWS]";
    mes "Happy Holidays! The reinboos are out in force this year and many are trying to get their hats right away.";
    next;
    mes "There seems to be a rumor going around of a three headed mouboo showing up again. Maybe you can catch it or maybe it's a myth. Someone let me know.";
    next;
    mes "Also there are rumors that the Wizards Council of Tulimshar may be accepting donations for a rumored trip to open the road to Gispaa soon. I'm thinking that they may need some brave adventurers to help them out when they get started.";
    next;
    mes "Kylian with a business liscence in his hand has said he wants to make a delivery business and is buying the shop across the street near the Wizard's Rest.";
    next;
    mes "WANT ADS";
    next;
    mes "Wushin is looking for people interested in becoming developers so if you are interested give him a shout at freenode #themanaworld-dev or just meet him in game.";
    next;
    mes "This has been Skragar Cinderweb with the news. As always if you have a story or person you'd like me to interview type up a message to Skragar Cinderweb on the forums.";
    goto L_Return;

L_NotEnoughMoney:
    mes "[" + @npcname$ + "]";
    mes "\"I'm sure it's an oversight but you don't have enough money.\"";
    goto L_Return;

L_Return:
    return;
}
