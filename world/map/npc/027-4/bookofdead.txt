// Entrance to Underworld

027-4.gat,108,86,0|script|The Book of Dead|400,0,0
{
    // No Cheaters
    if isin("027-4.gat",102,86,112,92)
        goto L_BookOfDead;
    end;

L_BookOfDead:
    // Need Ink and Quills from sages
    if ( (countitem("QuillOfBinding") == 0) || (countitem("BloodInk") == 0) )
        goto L_NoInkQuill;
    mes "It appears to be a book with a list of names.";
    mes "Do you wish to use the Quill of Binding and Blood Ink, do you wish to inscribe your name?";
    menu
       "Yes.", L_Under,
       "No thank you", L_Close;

L_NoInkQuill:
    mes "You have nothing to write with.";
    mes "I think Alacrius mentioned something about a Quill and Ink.";
    goto L_Close;

L_Under:
    warp "070-1.gat",30,93;
    goto L_Close;

L_Close:
    close;
}
