function|script|rubberBat
{
    if (gettimetick(2) - @lastbat < 2) goto L_Return;
    misceffect 403;
    set @lastbat, gettimetick(2);
    goto L_Return;

L_Return:
    getitem "RubberBat", 1;
    debugmes "Check";
    if ($DOOMSDAY == 3)
        addtimer 10, "#TMWFinalExam::OnRubberBat";
    return;
}
