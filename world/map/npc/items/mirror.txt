function|script|useMirror|,
{
  goto L_Rand;
  
  L_Rand:
    set @style, rand(0,19);
    set @color, rand(0,11);
    if((getlook(LOOK_HAIR_STYLE) == @style) || (getlook(LOOK_HAIR_COLOR) == @color))
      goto L_Rand;
    goto L_Proceed;
  
  L_Proceed:
    if(@fixedMirror)
      setlook LOOK_HAIR_STYLE, @style;
    setlook LOOK_HAIR_COLOR, @color;
    if(!@fixedMirror)
      getitem "SilverMirror", 1;
    set @fixedMirror, 0;
  return;
}
