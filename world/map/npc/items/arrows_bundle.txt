function|script|UnbundleArrows
{
    setarray .@arrows[0], Arrow, IronArrow, BoneArrows, SilverArrow;
    set .@arrow, .@arrows[rand(getarraysize(.@arrows))]; // Pick random arrow from list
    set .@qty, rand(100, 400);
    misceffect FX_GETITEM, strcharinfo(0);
    getitem .@arrow, .@qty; // No inv checks to keep it simple.
    return; // In worst case arrows just appear on floor nearby, ok for cheap item
}
