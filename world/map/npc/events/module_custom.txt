function|script|eventModule_customMenu|,
{
  next;
  mes "[Module: Custom]";
  mes "Starting this event opens the event portals and does.... nothing else.";
  menu
    "Start the event.", L_Start,
    "Abort.", L_Abort;
  
  L_Start:
    set $@event_ongoing_id$, "custom";
    set $@event_ongoing, 1;
    cmdothernpc "#EventPortals", "Open";
    mes "Event started.";
    close;
  
  L_Abort:
    return;
}

function|script|eventModule_customOngoingMenu|,
{
  next;
  mes "[Module: Custom]";
  menu
    "End this event.", L_Stop,
    "Do nothing", L_End;
  
  L_Stop:
    cmdothernpc "#EventPortals", "Close";
    callfunc "eventJanitorGlobal";
    close;
  
  L_End:
    close;
}
