// Dock

koga-candor.gat,39,29,0|script|#KogaCandorExit|45,0,1,
{
    mes "Disembark the ferry for "+$@CandorDocks$[$@CandorCurrentDock]+"?";
    menu
        "Yes.", L_Leave,
        "Nevermind", L_Close;

L_Leave:
    if ($@CandorCurrentDock == 0)
        goto L_Candor;
    if ($@CandorCurrentDock == 1)
        goto L_Tulimshar;
    goto L_Close;

L_Tulimshar:
    warp "022-1.gat", 60, 33;
    goto L_Close;

L_Candor:
    warp "029-1.gat", 25, 37;
    goto L_Close;

L_Close:
    close;
}
