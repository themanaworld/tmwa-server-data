//This barrier is for checking whether the player went outside after
//progressing with the cat quest.
//See katze.txt for more info

015-1,59,32,0|script|#CatOutsideBarrier|32767,1,1
{
    //handle each state after leaving the cat alone
    if (QL_CAT_EARS == 11) goto L_LeaveToProgress; //has stuff
    if (QL_CAT_EARS == 8) goto L_LeaveToProgress; //has log
    if (QL_CAT_EARS == 4) goto L_LeaveToProgress; //has food
    if (QL_CAT_EARS == 2) goto L_LeaveToProgress; //has milk

    end;

L_LeaveToProgress:
    set QL_CAT_EARS, QL_CAT_EARS + 1;
    end;
}
