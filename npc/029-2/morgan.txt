    callfunc "ClearVariables";

    return;
}
function	script	SetMorganMask	{
    STARTAREA = (STARTAREA & ~(NIBBLE_3_MASK) | (@morgan_tmp << NIBBLE_3_SHIFT));
    @morgan_tmp = 0;
    return;
}
