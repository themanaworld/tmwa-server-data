
function	script	Inn	{
    mes "[" + @npcname$ + "]";
    mes "\"Would you like to rest? It's only " + @Cost + " gp.\"";
    next;
    menu
        "Yes", L_Next,
        "No", L_No;

L_Next:
    if (Zeny < @Cost)
        goto L_NoMoney;
    Zeny = Zeny - @Cost;
    heal 10000, 10000;

    mes "[" + @npcname$ + "]";
    mes "\"Sleep well!\"";
    close;

L_No:
    mes "[" + @npcname$ + "]";
    mes "\"See you.\"";
    close;

L_NoMoney:
    mes "[" + @npcname$ + "]";
    mes "\"You don't have enough money to stay here.\"";
    close;
}
